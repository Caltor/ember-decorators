    public action() source 

    import {action} from 'ember-decorators/object'

    Decorator that turns the target function into an Action

    Adds an actions object to the target object and creates a passthrough function that calls the original. This means the function still exists on the original object, and can be used directly.

    import Component from '@ember/component';
import { action } from 'ember-decorators/object';

export default class ActionDemoComponent extends Component {
  @action
  foo() {
    // do something
  }
}

    <button onclick={{action "foo"}}>Execute foo action</button>

    public computed(propertyNames: ...String) source 

    import {computed} from 'ember-decorators/object'

    Decorator that turns a function into a computed property.

    With Dependent Keys

    The values of the dependent properties are passed as arguments to the function. You can also use property brace expansion.

    import EmberObject from '@ember/object';
import computed from 'ember-decorators/object';

export default class User extends EmberObject {
  someKey = 'foo';
  otherKey = 'bar';

  person = {
    firstName: 'John',
    lastName: 'Smith'
  };

  @computed('someKey', 'otherKey')
  foo(someKey, otherKey) {
    return `${someKey} - ${otherKey}`; // => 'foo - bar'
  }

  @computed('person.{firstName,lastName}')
  fullName(firstName, lastName) {
    return `${firstName} ${lastName}`; // => 'John Smith'
  }
}

    Without Dependent Keys

    Computed properties without dependent keys are cached for the whole life span of the object. You can only force a recomputation by calling notifyPropertyChange on the computed property.

    import EmberObject from '@ember/object';
import computed from 'ember-decorators/object';

export default class FooBar extends EmberObject {
  @computed
  foo() {
    // calculate stuff
    return stuff;
  }
}

    Getter and Setter

    import Component from '@ember/component';
import { setProperties } from ''@ember/object';
import computed from 'ember-decorators/object';

export default class UserProfileComponent extends Component {
  first = 'John';
  last = 'Smith';

  @computed('first', 'last')
  name = {
    get(first, last) {
      return `${first} ${last}`; // => 'John Smith'
    },

    set(value, first, last) {
      if (typeof value !== 'string' || !value.test(/^[a-z]+ [a-z]+$/i)) {
        throw new TypeError('Invalid name');
      }

      const [first, last] = value.split(' ');
      setProperties(this, { first, last });

      return value;
    }
  };
}

    NameTypeAttributeDescription
    propertyNames ...String

    List of property keys this computed is dependent on

    public observes(eventNames: ...String) source 

    import {observes} from 'ember-decorators/object'

    Decorator that wraps Ember.observer

    Triggers the target function when the dependent properties have changed

    import Component from '@ember/component';
import { observes } from 'ember-decorators/object';

export default class extends Component {
  @observes('foo')
  bar() {
    //...
  }
}

    Params:

    NameTypeAttributeDescription
    eventNames ...String

    Names of the events that trigger the function

    public readOnly() source 

    import {readOnly} from 'ember-decorators/object'

    Decorator that modifies a computed property to be read only.

    Usage:

    import Component from '@ember/component';
import { computed, readOnly } from 'ember-decorators/object';

export default class extends Component {
  @readOnly
  @computed('first', 'last')
  name(first, last) {
    return `${first} ${last}`;
  }
}
